合并两个有序数组

  1. 2023-03-23
  2. 387字
  3. 2 min read

https://leetcode.cn/problems/merge-sorted-array/

刷这个题主要想掌握的是js的基本语法和排序的基本知识与用法

  • 采用双指针方式，由于给出的数组已经排序，所以将两个数组看成队列，定义两个队首指针，每次取相较小的数放入中间数组
  • 时间复杂度为O(m+n)
  • 空间复杂度为O(m+n)
nums1 = [1,2,3,0,0,0];
m = 3;
nums2 = [2,5,6];
n = 3;

var merge = function(nums1,m,nums2,n){
    // 记录nums1当前指针的位置
    let p1 = 0 , p2 = 0;
    // 记录较小值
    var cur;
    const temp = new Array(m+n).fill(0)
    while(p1 < m || p2 < n){
        if(p1 === m){
            cur = nums2[p2++]
        }else if (p2 === n){
            cur = nums1[p1++] ;
        }else if (nums1[p1] < nums2[p2]){
            cur = nums1[p1++] ;
        }else cur = nums2[p2++]
        temp [p1 + p2 -1] = cur;
    }
    for ( let i = 0 ; i != m + n ; ++i ){
        nums1[i] = temp[i] ;
        console.log(nums1[i]) ;
    }
}

merge(nums1,m,nums2,n);

js里没有main函数

当把一个JavaScript文件交给JS引擎执行时，JS引擎就是从上到下逐条执行每条语句的，直到执行完所有代码。

虚拟main函数

var a = 1;
var b = 2;

function add(x, y) {
    var z = x + y;
    return z;
}

console.log(add(a, b));

JS引擎在程序开始执行前会把他包装成一个main函数

// 虚构的main函数
function main() {
    var a = 1;
    var b = 2;

    function add(x, y) {
        var z = x + y;
        return z;
    }

    console.log(add(a, b));
}

调用虚拟的main函数

main._current_scope_ = window; // 将全局作用域(对象)设为window
main.call(window) // 将this指向window
