void merge(int arr[], int temp[], int left, int mid, int right)
{
	int l_pos = left;
	int r_pos = mid + 1;
	int pos = left;
	
	while(l_pos <= mid && r_pos <= right)
	{
		if(arr[l_pos] < arr[r_pos])
			temp[pos++] = arr[l_pos++];
		else temp[pos++] = arr[r_pos++];
	}
	
	while(l_pos <= mid)
	{
		temp[pos++] = arr[l_pos++];
	}
	
	while(r_pos <= right)
	{
		temp[pos++] = arr[r_pos++];
	} 
	
	while(left <= right ){
		arr[left] = temp[left];
		left++;
	}
}
