归并排序

归并排序（Merge sort）是建立在归并操作上的一种有效、稳定的排序算法，该算法是采用分治法(Divide and Conquer）的一个非常典型的应用。将已有序的子序列合并，得到完全有序的序列；即先使每个子序列有序，再使子序列段间有序。若将两个有序表合并成一个有序表，称为二路归并。

归并排序适用于数据量大，并且对稳定性有要求的场景

c语言实现归并排序

#include<bits/stdc++.h>

void print_arr(int arr[],int n){
	for(int i = 0 ; i < n ; i ++ ){
		printf("%d ",arr[i]);
	}
}

void merge(int arr[], int temp[], int left, int mid, int right)
{
	int l_pos = left;
	int r_pos = mid + 1;
	int pos = left;
	
	while(l_pos <= mid && r_pos <= right)
	{
		if(arr[l_pos] < arr[r_pos])
			temp[pos++] = arr[l_pos++];
		else temp[pos++] = arr[r_pos++];
	}
	
	while(l_pos <= mid)
	{
		temp[pos++] = arr[l_pos++];
	}
	
	while(r_pos <= right)
	{
		temp[pos++] = arr[r_pos++];
	} 
	
	while(left <= right ){
		arr[left] = temp[left];
		left++;
	}
}

void msort(int arr[], int tempArr[], int left, int right) 
{
	if(left < right)
	{
		int mid = (left + right)/2;
//		递归分左边和右边递归划分
 		msort(arr, tempArr, left, mid);
 		msort(arr, tempArr, mid + 1, right);
// 		合并 
		merge(arr, tempArr, left, mid, right); 
	}	
}

void merge_sort(int arr[], int n){
	int* tempArr = (int*)malloc(n * sizeof(int));
	if(tempArr)
	{
		msort(arr, tempArr, 0, n - 1);
		free(tempArr);
	}else
	{
		printf("error");
	}
}

js实现归并排序

var printArr = function (arr, n) {
   for (var i = 0; i < n; i++) {
       console.log(arr[i]);
   }
}

var msort = function (arr, temp, left, mid, right) {
   // 左半区第一个未排序的元素下标
   var lef_pos = left;
   // 右半区第一个未排序的元素下标
   // var right_pos = 0;
   var right_pos = mid + 1;
   // 临时数组下标
   var pos = left;

   while (lef_pos <= mid && right_pos <= right)
   {
       if (arr[lef_pos] < arr[right_pos])
           temp[pos++] = arr[lef_pos++];
       else 
           temp[pos++] = arr[right_pos++];
   }

   // 分组排序完成后，如果还有，说明数组有剩余
   // 归并剩余的左半边
   while (lef_pos <= mid) {
       temp[pos++] = arr[lef_pos++];
   }

   while (right_pos <= right) {
       temp[pos++] = arr[right_pos++];
   }

   // 覆盖原来的无序数组
   while (left <= right) {
       arr[left] = temp[left];
       left++;
   }
   // printArr(temp,8);
}

var merge_sort = function (arr, temp, left, right) {
   // 递归拆分
   if (left < right) { //有两个以上元素,如果只有一个元素，那么left = right
       // 找中间点
       let mid = Math.floor((left + right) / 2);
       // 递归划分左半区域
       merge_sort(arr, temp, left, mid);
       // 递归划分右半区域
       merge_sort(arr, temp, mid + 1, right);
       // 合并已经排序的部分
       msort(arr, temp, left, mid, right)
   }
}


// 分配临时数组，进行排序
var merge = function (arr, n) {
   let temp = new Array(n).fill(0);
   // printArr(temp,n);
   merge_sort(arr, temp, 0, n - 1);
}
